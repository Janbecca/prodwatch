---- 基础配置层 ----
-- 项目表
CREATE TABLE project (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE,
    product_category TEXT,
    description TEXT,
    our_brand_id INTEGER,
    status TEXT,
    is_active INTEGER DEFAULT 0,
    refresh_mode TEXT,
    refresh_cron TEXT,
    last_refresh_at DATETIME,
    created_at DATETIME,
    updated_at DATETIME,
    deleted_at DATETIME,
    FOREIGN KEY (our_brand_id) REFERENCES brand(id)
);

-- 品牌表
CREATE TABLE brand (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE,
    alias TEXT,
    category TEXT,
    created_at DATETIME
);

-- 平台表
CREATE TABLE platform (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code TEXT NOT NULL UNIQUE,
    name TEXT NOT NULL,
    is_enabled INTEGER DEFAULT 1,
    created_at DATETIME
);

-- 项目-品牌
CREATE TABLE project_brand (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    brand_id INTEGER NOT NULL,
    is_core_brand INTEGER,
    created_at DATETIME,
    UNIQUE(project_id, brand_id),
    FOREIGN KEY (project_id) REFERENCES project(id),
    FOREIGN KEY (brand_id) REFERENCES brand(id)
);

-- 项目-平台
CREATE TABLE project_platform (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    platform_id INTEGER NOT NULL,
    created_at DATETIME,
    UNIQUE(project_id, platform_id),
    FOREIGN KEY (project_id) REFERENCES project(id),
    FOREIGN KEY (platform_id) REFERENCES platform(id)
);

-- 项目关键词
CREATE TABLE project_keyword (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    keyword TEXT NOT NULL,
    keyword_type TEXT,
    weight INTEGER,
    is_enabled INTEGER DEFAULT 1,
    created_at DATETIME,
    FOREIGN KEY (project_id) REFERENCES project(id)
);


---- 采集任务层 ----
-- 爬虫任务
CREATE TABLE crawl_job (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    job_type TEXT,
    trigger_source TEXT,
    schedule_type TEXT,
    schedule_expr TEXT,
    status TEXT,
    started_at DATETIME,
    ended_at DATETIME,
    created_by TEXT,
    error_message TEXT,
    FOREIGN KEY (project_id) REFERENCES project(id)
);

-- 爬虫任务目标
CREATE TABLE crawl_job_target (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    crawl_job_id INTEGER NOT NULL,
    platform_id INTEGER,
    brand_id INTEGER,
    keyword TEXT,
    FOREIGN KEY (crawl_job_id) REFERENCES crawl_job(id),
    FOREIGN KEY (platform_id) REFERENCES platform(id),
    FOREIGN KEY (brand_id) REFERENCES brand(id)
);

---- 原始数据层 ----
-- 原始帖子
CREATE TABLE post_raw (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    crawl_job_id INTEGER,
    platform_id INTEGER NOT NULL,
    brand_id INTEGER,
    external_post_id TEXT,
    author_name TEXT,
    title TEXT,
    content TEXT,
    post_url TEXT,
    publish_time DATETIME,
    crawled_at DATETIME,
    like_count INTEGER,
    comment_count INTEGER,
    share_count INTEGER,
    view_count INTEGER,
    raw_payload TEXT,
    dedup_key TEXT,
    created_at DATETIME,
    UNIQUE(project_id, platform_id, dedup_key),
    FOREIGN KEY (project_id) REFERENCES project(id),
    FOREIGN KEY (crawl_job_id) REFERENCES crawl_job(id),
    FOREIGN KEY (platform_id) REFERENCES platform(id),
    FOREIGN KEY (brand_id) REFERENCES brand(id)
);

-- 帖子-品牌关系（多对多）
CREATE TABLE post_brand_relation (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id INTEGER NOT NULL,
    brand_id INTEGER NOT NULL,
    relation_type TEXT,
    FOREIGN KEY (post_id) REFERENCES post_raw(id),
    FOREIGN KEY (brand_id) REFERENCES brand(id)
);


---- 分析结果层 ----
-- 清洗结果
CREATE TABLE post_clean_result (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id INTEGER NOT NULL,
    is_valid INTEGER,
    invalid_reason TEXT,
    clean_text TEXT,
    language TEXT,
    analyzed_at DATETIME,
    FOREIGN KEY (post_id) REFERENCES post_raw(id)
);

-- 情感分析
CREATE TABLE post_sentiment_result (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id INTEGER NOT NULL,
    sentiment TEXT,
    sentiment_score REAL,
    emotion_intensity REAL,
    model_version TEXT,
    analyzed_at DATETIME,
    FOREIGN KEY (post_id) REFERENCES post_raw(id)
);

-- 水军识别
CREATE TABLE post_spam_result (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id INTEGER NOT NULL,
    spam_label TEXT,
    spam_score REAL,
    analyzed_at DATETIME,
    FOREIGN KEY (post_id) REFERENCES post_raw(id)
);

-- 关键词识别
CREATE TABLE post_keyword_result (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id INTEGER NOT NULL,
    keyword TEXT,
    keyword_type TEXT,
    confidence REAL,
    analyzed_at DATETIME,
    FOREIGN KEY (post_id) REFERENCES post_raw(id)
);

-- 功能点分析
CREATE TABLE post_feature_result (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id INTEGER NOT NULL,
    feature_name TEXT,
    feature_sentiment TEXT,
    confidence REAL,
    analyzed_at DATETIME,
    FOREIGN KEY (post_id) REFERENCES post_raw(id)
);


----  统计聚合层 ----
-- 日统计
CREATE TABLE daily_metric (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    brand_id INTEGER,
    platform_id INTEGER,
    stat_date DATE,
    total_post_count INTEGER,
    valid_post_count INTEGER,
    spam_post_count INTEGER,
    positive_count INTEGER,
    neutral_count INTEGER,
    negative_count INTEGER,
    avg_sentiment_score REAL,
    total_like_count INTEGER,
    total_comment_count INTEGER,
    total_share_count INTEGER,
    keyword_count INTEGER,
    created_at DATETIME,
    UNIQUE(project_id, brand_id, platform_id, stat_date),
    FOREIGN KEY (project_id) REFERENCES project(id),
    FOREIGN KEY (brand_id) REFERENCES brand(id),
    FOREIGN KEY (platform_id) REFERENCES platform(id)
);

-- 关键词统计
CREATE TABLE daily_keyword_metric (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    brand_id INTEGER,
    platform_id INTEGER,
    stat_date DATE,
    keyword TEXT,
    hit_count INTEGER,
    created_at DATETIME,
    FOREIGN KEY (project_id) REFERENCES project(id)
);

-- 功能点统计
CREATE TABLE daily_feature_metric (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    brand_id INTEGER,
    stat_date DATE,
    feature_name TEXT,
    mention_count INTEGER,
    positive_count INTEGER,
    neutral_count INTEGER,
    negative_count INTEGER,
    FOREIGN KEY (project_id) REFERENCES project(id)
);

---- 报告中心 ----
-- 报告主表
CREATE TABLE report (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL,
    title TEXT,
    report_type TEXT,
    data_start_date DATE,
    data_end_date DATE,
    status TEXT,
    summary TEXT,
    content_markdown TEXT,
    created_by TEXT,
    created_at DATETIME,
    updated_at DATETIME,
    FOREIGN KEY (project_id) REFERENCES project(id)
);

-- 报告配置
CREATE TABLE report_config (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    report_id INTEGER NOT NULL,
    platform_ids TEXT,
    brand_ids TEXT,
    keywords TEXT,
    include_sentiment INTEGER,
    include_trend INTEGER,
    include_topics INTEGER,
    include_feature_analysis INTEGER,
    include_spam INTEGER,
    include_competitor_compare INTEGER,
    include_strategy INTEGER,
    FOREIGN KEY (report_id) REFERENCES report(id)
);

-- 报告引用证据
CREATE TABLE report_evidence (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    report_id INTEGER NOT NULL,
    post_id INTEGER NOT NULL,
    section_name TEXT,
    quote_reason TEXT,
    sentiment TEXT,
    spam_label TEXT,
    created_at DATETIME,
    FOREIGN KEY (report_id) REFERENCES report(id),
    FOREIGN KEY (post_id) REFERENCES post_raw(id)
);


---- 索引 ----
CREATE INDEX idx_post_project ON post_raw(project_id);
CREATE INDEX idx_post_time ON post_raw(publish_time);
CREATE INDEX idx_post_platform ON post_raw(platform_id);

CREATE INDEX idx_sentiment_post ON post_sentiment_result(post_id);
CREATE INDEX idx_spam_post ON post_spam_result(post_id);

CREATE INDEX idx_daily_metric ON daily_metric(project_id, stat_date);
